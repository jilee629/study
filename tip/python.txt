# PIP 설치
$ sudo apt-get install -y python3-pip


# pip install 시 에러발생하면
$ python3 -m pip config set global.break-system-packages true


# pip install 시 path 에러발생하면 ~/.bashrc 에 pip path 추가하기
export PATH="$PATH:/home/ubuntu/.local/bin"


# package 설치
$ pip install --upgrade(-U) <package>


# 설치된 package에서 업데이트된 리스트 확인
pip list --outdated(-o)


# pipupgrade
$ pip install pipupgrade 
$ pipupgrade --yes --latest --verbose
$ pipupgrade --yes --latest --verbose --ignore dbus-python --ignore PyGObject


# pip error 발생시
python3 -m pip config set global.break-system-packages true
